1069 void HelloTriangle::transitionImageLayout(
1070  VkImage image, VkFormat format,
1071  VkImageLayout old_layout, VkImageLayout new_layout) {
1072  VkCommandBuffer command_buffer = beginSignalCommand();
1073  //
1074  VkImageMemoryBarrier barrier{};
1075  barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
1076  barrier.oldLayout = old_layout;
1077  barrier.newLayout = new_layout;
1078  barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
1079  barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
1080  barrier.image = image;
1081  barrier.subresourceRange.baseMipLevel = 0;
1082  barrier.subresourceRange.levelCount = 1;
1083  barrier.subresourceRange.baseArrayLayer = 0;
1084  barrier.subresourceRange.layerCount = 1;
1085 
1086  VkPipelineStageFlags source_stage;
1087  VkPipelineStageFlags dst_stage;
1088  //
1089  if (new_layout ==
1090  VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL) {
1091  barrier.subresourceRange.aspectMask =
1092  VK_IMAGE_ASPECT_DEPTH_BIT;
1093  if (hasStencilSupport(format)) {
1094  barrier.subresourceRange.aspectMask |=
1095  VK_IMAGE_ASPECT_STENCIL_BIT;
1096  }
1097  } else {
1098  barrier.subresourceRange.aspectMask =
1099  VK_IMAGE_ASPECT_COLOR_BIT;
1100  }
1101 
1102  // pipeline 1
1103  if (old_layout == VK_IMAGE_LAYOUT_UNDEFINED &&
1104  new_layout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
1105 
1106  barrier.srcAccessMask = 0;
1107  barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
1108 
1109  source_stage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
1110  dst_stage = VK_PIPELINE_STAGE_TRANSFER_BIT;
1111  // pipeline 2
1112  } else if (old_layout ==
1113  VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL &&
1114  new_layout ==
1115  VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL) {
1116  barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
1117  barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
1118 
1119  source_stage = VK_PIPELINE_STAGE_TRANSFER_BIT;
1120  dst_stage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
1121  // pipeline 3
1122  } else if (
1123  old_layout == VK_IMAGE_LAYOUT_UNDEFINED &&
1124  new_layout ==
1125  VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL) {
1126  barrier.srcAccessMask = 0;
1127  auto rwflag =
1128  VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_READ_BIT |
1129  VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_WRITE_BIT;
1130  barrier.dstAccessMask = rwflag;
1131  source_stage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
1132  dst_stage = VK_PIPELINE_STAGE_EARLY_FRAGMENT_TESTS_BIT;
1133  } else {
1134  throw std::invalid_argument(
1135  "unsupported layout transition");
1136  }
1137  vkCmdPipelineBarrier(command_buffer, source_stage,
1138  dst_stage, 0, 0, nullptr, 0, nullptr,
1139  1, &barrier);
1140  endSignalCommand(command_buffer);
1141 }
